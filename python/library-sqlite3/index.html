from collections import namedtuple

def namedtuple_factory(cursor, row):
    """Returns sqlite rows as named tuples."""
    fields = [col[0] for col in cursor.description]
    x = namedtuple("x", fields)
    return x(*row)
db.row_factory = namedtuple_factory
c = db.cursor()
