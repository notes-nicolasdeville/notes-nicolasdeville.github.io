import re

for row in ws:
    count_row += 1
    first = row[0].value.strip()
    command = [s for s in re.split("([A-Z][^A-Z]*)", first) if s][0].strip()
    description = re.findall('[A-Z][^A-Z]*', first)[0]
    print(f"{command=}")
    print(f"{description=}")
    print()
